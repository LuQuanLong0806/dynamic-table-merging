```html
  </script>

  <script src="...vue.min.js"></script>
  <script src="...iview.min.js"></script>
```

### 2.1 为什么用 `<script type="text/x-template">`

Vue 2 支持三种模板方式：
1. **字符串模板** `template: '<div>...</div>'` — 需要大量转义引号，易出错
2. **inline-template** — 模板写在 DOM 中，解析时机不可控
3. **x-template** `<script type="text/x-template" id="app-tpl">` — **推荐**：浏览器不执行，Vue 通过 `template: '#app-tpl'` 引用，无需转义

## 四、核心算法 — buildGrid

这是整个引擎最核心的函数，负责将扁平的 cells 数组转换为二维网格结构供模板渲染。

### 4.1 完整源码

```js
function buildGrid(cells) {
  // ── 第1步：建立 id → cell 映射表 ──
  var cellMap = {};
  cells.forEach(function(c) { cellMap[c.id] = c; });

  // ── 第2步：计算网格尺寸 ──
  var maxC = 0, maxR = 0;
  cells.forEach(function(c) {
    var xr = parseRange(c.x), yr = parseRange(c.y);
    if (xr.end > maxC) maxC = xr.end;
    if (yr.end > maxR) maxR = yr.end;
  });

  // ── 第3步：初始化矩阵 ──
  // G[r][c] = 放置的 entry 对象 或 null
  // O[r][c] = true 表示被合并占据（skip），false 表示空闲
  var G = [], O = [];
  for (var r = 0; r < maxR; r++) {
    G[r] = [];
    O[r] = [];
    for (var c = 0; c < maxC; c++) {
      G[r][c] = null;
      O[r][c] = false;
    }
  }

  // ── 第4步：按合并面积从大到小排序 ──
  // 原因：大合并必须先放置，小单元格后放置
  // 如果先放小单元格，大合并可能找不到完整空间
  var sorted = cells.slice().sort(function(a, b) {
    return parseRange(b.y).span * parseRange(b.x).span
         - parseRange(a.y).span * parseRange(a.x).span;
  });

  // ── 第5步：逐个放置 cell ──
  sorted.forEach(function(cell) {
    var xr = parseRange(cell.x), yr = parseRange(cell.y);
    var r0 = yr.start - 1;   // 转为 0-based 索引
    var c0 = xr.start - 1;
    var entry = { cell: cell, rowspan: yr.span, colspan: xr.span, alts: null };

    // 检查左上角位置是否已被占用
    if (G[r0][c0]) {
      // 已被占用 → 作为备选项(alternative)追加
      if (!G[r0][c0].alts) G[r0][c0].alts = [];
      G[r0][c0].alts.push(entry);
      return;  // 不标记 O 矩阵
    }

    // 未被占用 → 放入主 entry
    G[r0][c0] = entry;

    // 标记合并覆盖的所有位置
    for (var r = r0; r < r0 + yr.span && r < maxR; r++)
      for (var c = c0; c < c0 + xr.span && c < maxC; c++)
        if (!(r === r0 && c === c0)) O[r][c] = true;
  });

  // ── 第6步：生成渲染用的行数组 ──
  var rows = [];
  for (var r = 0; r < maxR; r++) {
    var cols = [];
    for (var c = 0; c < maxC; c++) {
      // 情况1：被合并占据 → skip
      if (O[r][c]) {
        cols.push({ skip: true });
        continue;
      }
      // 情况2：空格子（无 cell）
      var e = G[r][c];
      if (!e) {
        cols.push({ skip: false, cell: null });
        continue;
      }
      // 情况3：有 cell
      var isCtrl = ['input','number','select','radio','checkbox','date']
                   .indexOf(e.cell.type) >= 0;
      var cls = [];
      if (e.cell.type === 'text' && (c === 0 || c === 2) && /<b>/.test(e.cell.content || ''))
        cls.push('cell-group');

      cols.push({
        skip: false,
        cell: e.cell,
        rowspan: e.rowspan,
        colspan: e.colspan,
        alts: e.alts || [],
        isLabel: e.cell.type === 'label',
        isText: e.cell.type === 'text',
        isControl: isCtrl,
        isFormula: e.cell.type === 'formula',
        cls: cls.join(' ')
      });
    }
    rows.push(cols);
  }

  return { rows: rows, cellMap: cellMap };
}
```

### 4.2 算法图解

假设有以下 cells：

```
cell A: x:"1", y:"1,3" (3行合并, 面积=3)
cell B: x:"2,3", y:"1"  (2列合并, 面积=2)
cell C: x:"2", y:"2"
cell D: x:"3", y:"2"
```

排序后放置顺序：A(面积3) → B(面积2) → C(面积1) → D(面积1)

```
放置 A 后：         放置 B 后：         放置 C、D 后：
┌────┬────┬────┐    ┌────┬────┬────┐    ┌────┬────┬────┐
│ A  │    │    │    │ A  │  B │  B │    │ A  │  B │  B │
│    ├────┼────┤    │    ├────┴────┤    │    ├────┼────┤
│    │    │    │    │    │         │    │    │  C │  D │
│    ├────┼────┤    │    │         │    │    ├────┼────┤
│    │    │    │    │    │         │    │    │    │    │
└────┴────┴────┘    └────┴─────────┘    └────┴────┴────┘

O 矩阵：           O 矩阵：           O 矩阵：
F F F              F T T              F T T
T F F              T T T              T F F
T F F              T T T              T F F
```

### 4.3 备选项处理

当两个 cell 的 `(x, y)` 左上角坐标完全相同时：

```
cell d_energy: x:"4", y:"5", type:"number"  (先放置，占主位)
cell d_none:   x:"4", y:"5", type:"text"    (后放置，左上角已被占)
  → d_none 追加到 d_energy 的 entry.alts 数组中
```

模板渲染时，主 cell 和所有 alts 在同一个 `<td>` 内，通过 `v-show` 控制切换。

### 4.4 返回值结构

```js
{
  rows: [
    // 第1行
    [
      { skip:false, cell:{id:"h1",...}, rowspan:1, colspan:1, alts:[], isLabel:true, ... },
      { skip:false, cell:{id:"h2",...}, rowspan:1, colspan:1, alts:[], isLabel:true, ... },
      ...
    ],
    // 第2行
    [
      { skip:false, cell:{id:"a1",...}, rowspan:2, colspan:1, alts:[], isText:true, ... },
      { skip:false, cell:{id:"a2",...}, rowspan:1, colspan:1, alts:[], isText:true, ... },
      ...
    ],
    ...
  ],
  cellMap: {
    "h1": cellObj,
    "c8": cellObj,
    ...
  }
}
```

## 五、工具函数

### 5.1 parseRange

将坐标字符串解析为数值范围：

```js
function parseRange(v) {
  var p = String(v).split(',').map(Number);
  return {
    start: Math.min.apply(null, p),   // 起始值
    end:   Math.max.apply(null, p),   // 结束值
    span:  Math.max.apply(null, p) - Math.min.apply(null, p) + 1  // 跨度
  };
}
```

示例：
- `parseRange("3")` → `{start:3, end:3, span:1}`
- `parseRange("2,7")` → `{start:2, end:7, span:6}`
- `parseRange("3,5")` → `{start:3, end:5, span:3}`

### 5.2 buildFormModel

从 cells 数组构建表单模型对象 `fm`，所有可交互控件的值都存在这里：

```js
function buildFormModel(cells) {
  var m = {};
  cells.forEach(function(c) {
    if (c.type === 'number')
      m[c.id] = (c.value != null && c.value !== '') ? Number(c.value) : null;
    else if (c.type === 'checkbox')
      m[c.id] = Array.isArray(c.value) ? c.value : [];
    else if (['input','select','radio','date'].indexOf(c.type) >= 0)
      m[c.id] = c.value != null ? c.value : '';
  });
  return m;
}
```

初始值规则：

| type | 无 value 时的初始值 | 有 value 时的处理 |
|------|---------------------|-------------------|
| `number` | `null` | `Number(value)` |
| `checkbox` | `[]` | 原样使用（必须为数组） |
| `input`/`select`/`radio`/`date` | `''` | 原样使用 |
| `label`/`text`/`formula` | 不初始化（不参与表单） | — |

## 六、Vue 实例完整设计

### 6.1 挂载与模板

```js
var app = new Vue({
  el: '#app',
  template: '#app-tpl',  // 引用 <script type="text/x-template" id="app-tpl">

```

> **关键**：使用 `var app` 声明，使实例可被外部代码访问，用于接口回显 `app.loadData(data)`。

### 6.2 data 初始化

```js
data: function() {
  var grid = buildGrid(TABLE_DATA.cells);
  var fm = buildFormModel(TABLE_DATA.cells);

  // 初始化 disabledState：遍历所有 cell，disabled:true 的加入
  var ds = {};
  TABLE_DATA.cells.forEach(function(c) { if (c.disabled) ds[c.id] = true; });

  // 初始化 hiddenState：遍历所有 cell，_hidden:true 的加入
  var hs = {};
  TABLE_DATA.cells.forEach(function(c) { if (c._hidden) hs[c.id] = true; });

  return {
    config: TABLE_DATA,        // 原始 JSON 配置（引用）
    gridRows: grid.rows,       // 二维网格数组，模板 v-for 遍历
    cellMap: grid.cellMap,     // {id: cellObj} 映射表
    fm: fm,                    // 表单模型 {cellId: value}
    disabledState: ds,         // {cellId: true} 禁用状态
    hiddenState: hs,           // {cellId: true} 隐藏状态
    submitDisabled: false,     // 提交按钮禁用状态
    formulaResults: {},        // {cellId: 计算结果} 公式缓存
    colWidths: ['6%','3%','12%','12%','6%','4%','7%','7%','7%','6%','7%','12%','12%']
  };
}
```

> **为什么用 `data: function()` 而非 `data: {}`**：工厂函数确保每次创建实例时重新计算，避免多实例共享引用。

### 6.3 computed 属性

```js
computed: {
  formRules: function() {
    var R = {}, self = this;
    this.config.cells.forEach(function(c) {
      // 只为有 validation 且非隐藏的控件生成规则
      if (c.validation && !c._hidden) {
        R[c.id] = self.buildRules(c);
      }
    });
    return R;
    // 返回格式: { "c8": [{required:true,...}, {type:"number",min:0,...}], ... }
  }
}
```

iView Form 通过 `<i-form :rules="formRules">` 绑定，`formRules` 的 key 必须与 `<form-item :prop="cellId">` 的 prop 一致，value 是 async-validator 规则数组。

## 七、完整模板渲染逻辑

### 7.1 模板总体结构

```html
<div class="table-wrapper">
  <h2>{{config.title}}</h2>
  <i-form ref="mainForm" :model="fm" :rules="formRules">
    <table>
      <colgroup>...</colgroup>
      <tr v-for="(row, ri) in gridRows" :key="ri">
        <template v-for="(entry, ci) in row">
          <!-- 渲染逻辑分支 -->
        </template>
      </tr>
    </table>
  </i-form>
  <div class="btn-bar">
    <i-button type="primary" :disabled="submitDisabled" @click="handleSubmit">提交</i-button>
    <i-button @click="handleReset" style="margin-left:8px">重置</i-button>
  </div>
</div>
```

### 7.2 iView Form 绑定要点

```html
<i-form ref="mainForm" :model="fm" :rules="formRules">
```

- `:model="fm"` — 表单数据对象，async-validator 从 `fm[prop]` 取值校验
- `:rules="formRules"` — 校验规则，key 是 cell id，value 是规则数组

每个控件必须包裹在 `<form-item :prop="cellId">` 中才能触发校验：

```html
<form-item v-else-if="item.isControl" :prop="item.cell.id">
  <!-- 控件 -->
</form-item>
```

### 7.3 列宽控制

```html
<colgroup>
  <col v-for="(w, i) in colWidths" :key="i" :style="{width: w}">
</colgroup>
```

`colWidths` 是一个百分比字符串数组，每个元素对应一列的宽度。

### 7.4 单元格渲染分支

对每个 `entry`，按优先级判断：

```
entry.skip === true
  → 不渲染 <td>（被合并占据的格子，HTML table 自动跳过）

entry.isLabel === true
  → 渲染 <th>（表头）
  → <th colspan="entry.colspan" rowspan="entry.rowspan" v-html="entry.cell.content">

entry 有 cell（非 skip、非 label）
  → 渲染 <td>
  → <td colspan="entry.colspan" rowspan="entry.rowspan">
  → 内部遍历 getItems(entry) 处理主 cell + 备选项

entry.cell === null
  → 渲染空 <td></td>（空格子）
```

### 7.5 getItems — 主 cell 与备选项

```js
getItems: function(e) {
  // 主 cell
  var list = [{
    cell: e.cell,
    isText: e.isText,
    isControl: e.isControl,
    isFormula: e.isFormula
  }];
  // 备选项
  (e.alts || []).forEach(function(a) {
    list.push({
      cell: a.cell,
      isText: a.cell.type === 'text',
      isControl: ['input','number','select','radio','checkbox','date'].indexOf(a.cell.type) >= 0,
      isFormula: a.cell.type === 'formula'
    });
  });
  return list;
}
```

模板中遍历 `getItems(entry)` 返回的列表，每个 item 用 `v-show="!isHidden(item.cell.id)"` 控制显隐。

### 7.6 每个 item 的渲染逻辑

```
item.isText === true
  → <span v-html="item.cell.content || ''"></span>

item.isFormula === true
  → <span class="formula-val">{{ formulaResults[item.cell.id] }}</span>

item.isControl === true
  → <form-item :prop="item.cell.id">
    → 根据 item.cell.type 渲染对应控件
```

## 八、控件渲染详细实现

### 8.1 input（文本输入框）

```html
<i-input v-if="item.cell.type==='input'"
  :value="fm[item.cell.id]"
  @input="val => setVal(item.cell.id, val)"
  :placeholder="item.cell.placeholder || item.cell.content || ''"
  :disabled="isDisabled(item.cell.id)"
  @on-blur="onBlur(item.cell.id)" />
```

| 属性/事件 | 说明 |
|-----------|------|
| `:value="fm[id]"` | 绑定值（非 v-model，原因见下文） |
| `@input` | iView Input 的输入事件，触发 `setVal` |
| `:placeholder` | 优先用 placeholder，fallback 到 content |
| `:disabled` | 受 `disabledState[id]` 控制 |
| `@on-blur` | 失焦时触发校验 |

### 8.2 number（数字输入框）

```html
<input-number v-if="item.cell.type==='number'"
  :value="fm[item.cell.id]"
  @on-change="val => setVal(item.cell.id, val)"
  :placeholder="item.cell.placeholder || ''"
  :min="getNum(item.cell, 'min')"
  :max="getNum(item.cell, 'max')"
  :disabled="isDisabled(item.cell.id)"
  style="width:100%" />
```

| 属性/事件 | 说明 |
|-----------|------|
| `@on-change` | InputNumber 的值变化事件（注意：与 Input 的 @input 不同） |
| `:min` / `:max` | 从 validation.rules 中提取，控制输入范围 |
| `style="width:100%"` | InputNumber 默认宽度不够，必须 100% |

`getNum` 方法从 validation 中提取 min/max：

```js
getNum: function(c, k) {
  var v = c.validation;
  if (!v) return undefined;
  var r;
  if (v.rules) v.rules.forEach(function(x) { if (x[k] != null) r = x[k]; });
  if (r == null && v[k] != null) r = v[k];
  return r;
}
```

### 8.3 select（下拉选择）

## 九、值绑定机制详解

### 9.1 为什么不用 v-model

iView 4.x 的组件在使用 v-model 时存在以下问题：

1. **InputNumber 空值问题**：清空输入框时 v-model 收到 `undefined`，而非 `null`，导致 async-validator 的 `type:"number"` 校验异常
2. **Select clearable 问题**：清空选择时 v-model 收到 `undefined`，`fm[id]` 变为 undefined，Vue 2 无法检测到变化
3. **响应式丢失**：Vue 2 的响应式系统无法检测到直接对对象属性赋 `undefined` 的变化

### 9.2 setVal 统一入口

```js
setVal: function(id, val) {
  // number 类型：空值统一为 null
  var cell = this.cellMap[id];
  if (cell && cell.type === 'number') {
    val = (val === '' || val === undefined || val === null) ? null : Number(val);
  }
  // 用 $set 保证 Vue 2 响应式
  this.$set(this.fm, id, val);

  // 立即清除校验红框
  var self = this;
  this.$nextTick(function() {
    var fi = self.$refs.mainForm.fields.find(function(f) { return f.prop === id; });
    if (fi) fi.validateState = '';
  });
}
```

关键点：
- **`this.$set(this.fm, id, val)`**：Vue 2 无法检测到直接添加新属性，必须用 `$set`
- **空值转 null**：number 类型空值统一为 `null`，确保 async-validator 正确判断
- **清除校验**：用户填值后立即清除红框，避免已填值仍显示错误提示

### 9.3 事件流

```
用户操作控件
  │
  ├─ input类型: @input → setVal(id,val) → $set(fm,id,val)
  ├─ number:    @on-change → setVal(id,val) → $set(fm,id,val)
  ├─ select:    @on-change → setVal + onChange(id)
  ├─ radio:     @on-change → setVal + onChange(id)
  ├─ checkbox:  @on-change → setVal + onChange(id)
  └─ date:      @on-change → setVal + onChange(id)

onChange(id) 的调用链：
  → handleLinkage(id)   // 查找并执行联动规则
  → recalc()            // 重算所有公式
  → validateField(id)   // 触发单字段校验

onBlur(id) 的调用链：
  → validateField(id)   // 仅触发校验
```

## 十、联动引擎

### 10.1 handleLinkage

```js
handleLinkage: function(id) {
  var cell = this.cellMap[id];
  if (!cell || !cell.linkage) return;

  var val = String(this.fm[id]);  // 当前值转字符串
  var self = this;

  cell.linkage.forEach(function(rule) {
    if (String(rule.trigger) === val) {         // 匹配触发值
      rule.actions.forEach(function(a) {
        self.applyAction(a);                     // 执行每个动作
      });
    }
  });
}
```

> **字符串比较**：`fm[id]` 可能是 number/string，`trigger` 是 string，统一 `String()` 后比较。

### 10.2 applyAction

```js
applyAction: function(a) {
  var self = this, t = a.target;

  // ── 单目标动作 ──
  if (t) {
    switch(a.type) {
      case 'show':      this.$set(this.hiddenState, t, false);  break;
      case 'hide':      this.$set(this.hiddenState, t, true);   break;
      case 'enable':    this.$set(this.disabledState, t, false); break;
      case 'disable':   this.$set(this.disabledState, t, true);  break;
      case 'clear':     this.doClear(t);                         break;
      case 'setValue':  this.$set(this.fm, t, a.value || '');    break;
    }
    return;
  }

  // ── 多目标行绑定动作 ──
  var targets = a.rowTargets;
  if (!targets) return;
  targets.forEach(function(rt) {
    rt.cols.forEach(function(cid) {
      switch(a.type) {
        case 'rowEnable':  self.$set(self.disabledState, cid, false); break;
        case 'rowDisable': self.$set(self.disabledState, cid, true);  break;
        case 'rowClear':   self.doClear(cid);                         break;
      }
    });
  });
}
```

> **关于寻址机制**：联动是**纯 id 寻址**，与坐标无关。
> - 单目标动作通过 `target` 字段指定目标 cell 的 id
> - 多目标动作通过 `rowTargets[].cols` 指定多个目标 cell 的 id
> - `rowTargets[].row` 字段仅为人工可读注释，引擎不使用
> - 引擎通过 id 在 `disabledState`/`hiddenState`/`fm` 中查找对应控件
> - 这意味着联动的 target 可以指向表格中**任意位置**的控件，不要求同一行

### 10.3 doClear

```js
doClear: function(id) {
  if (!this.fm.hasOwnProperty(id)) return;
  var tc = this.cellMap[id];
  var empty = tc && tc.type === 'checkbox' ? []
            : tc && tc.type === 'number'   ? null
            : '';
  this.$set(this.fm, id, empty);
}
```

根据控件类型返回不同的空值：checkbox→`[]`，number→`null`，其他→`''`。

### 10.4 联动不会无限循环

联动只在 `onChange` 中触发，而 `setVal` 不会调用 `onChange`。所以联动动作（如 `clear` 调用 `doClear` → `$set`）不会再次触发联动。

## 十一、校验系统

### 11.1 buildRules — 规则转换

```js
buildRules: function(cell) {
  var v = cell.validation;
  if (!v) return [];

  var isNum = cell.type === 'number';
  // 自动推断 trigger
  var trigger = (isNum || cell.type === 'select' || cell.type === 'radio')
              ? 'change' : 'blur';

  // 格式1: rules 数组
  if (v.rules && v.rules.length) {
    return v.rules.map(function(r) {
      var rule = {};
      for (var k in r) rule[k] = r[k];        // 浅拷贝
      if (!rule.trigger) rule.trigger = trigger; // 自动补 trigger
      if (isNum && rule.required && !rule.validator && !rule.depends) {
        rule.type = rule.type || 'number';       // number 必填自动补 type
      }
      // depends 条件校验：表达式求值为 false 时跳过规则
      if (rule.depends) {
        var depExpr = rule.depends;
        var orig = {
          required: rule.required, type: rule.type,
          min: rule.min, max: rule.max, pattern: rule.pattern,
          len: rule.len, message: rule.message || '校验未通过'
        };
        rule.validator = (function(dep, o, isNumType) {
          return function(r, value, callback) {
            if (!evalDepends(dep, self.fm, self.cellMap)) {
              callback();  // 条件不满足 → 跳过
              return;
            }
            // 条件满足 → 按原始规则校验
            if (o.required) {
              var empty = value == null || value === '' ||
                (Array.isArray(value) && value.length === 0);
              if (empty) { callback(new Error(o.message)); return; }
            }
            if (o.type === 'number' && value != null && value !== '') {
              var n = Number(value);
              if (isNaN(n)) { callback(new Error(o.message)); return; }
              if (o.min != null && n < o.min) { callback(new Error(o.message)); return; }
              if (o.max != null && n > o.max) { callback(new Error(o.message)); return; }
            }
            if (o.pattern) {
              var re = typeof o.pattern === 'string' ? new RegExp(o.pattern) : o.pattern;
              if (!re.test(String(value))) { callback(new Error(o.message)); return; }
            }
            callback();
          };
        })(depExpr, orig, isNum);
        // 删除与 validator 冲突的字段
        delete rule.required; delete rule.type;
        delete rule.min; delete rule.max;
        delete rule.pattern; delete rule.len;
        delete rule.message; delete rule.depends;
      }
      return rule;
    });
  }

  // 格式2: 简写
  var rules = [];
  if (v.required)
    rules.push({
      required: true,
      message: v.message || '此项为必填',
      trigger: trigger,
      type: isNum ? 'number' : 'string'
    });
  if (v.pattern)
    rules.push({
      pattern: new RegExp(v.pattern),
      message: v.message || '格式不正确',
      trigger: trigger
    });
  return rules;
}
```

### 11.1.1 evalDepends — 条件表达式求值

独立的纯函数，将 depends 表达式中的 cell id 替换为 `fm` 中的当前值后求值：

```js
function evalDepends(expr, fm, cellMap) {
  // 按 id 长度降序排列，避免短 id 部分替换长 id
  var ids = Object.keys(cellMap).sort(function(a, b) {
    return b.length - a.length;
  });
  ids.forEach(function(id) {
    var v = fm[id];
    var r;
    if (v === null || v === undefined) r = 'null';
    else if (typeof v === 'number') r = String(v);
    else if (typeof v === 'string') r = "'" + v.replace(/'/g, "\\'") + "'";
    else r = 'null';
    expr = expr.replace(
      new RegExp('\\b' + id.replace(/[.*+?^${}()|[\]\\]/g, '\\$&') + '\\b', 'g'),
      r
    );
  });
  try {
    return new Function('"use strict"; return (' + expr + ')')();
  } catch(e) {
    return false;  // 表达式出错 → 安全跳过
  }
}
```

**替换规则**：

| fm 值类型 | 替换结果 | 示例 |
|-----------|---------|------|
| `null` / `undefined` | `null` | `bm_r42` → `null` |
| number | 数字字面量 | `bm_r42` → `2` |
| string | 带引号字符串 | `c4` → `'has'` |

**安全机制**：
- 引用不存在的 cell id → 不替换 → 求值失败 → 返回 `false`（跳过规则）
- 语法错误 → `catch` → 返回 `false`
- 按 id 长度降序替换，防止 `bm_r42` 被 `bm_r4` 部分匹配

**单元测试**：`test-depends.js`，51 个用例覆盖 depends、custom、accept/maxSize 全部场景。

### 11.1.2 evalCustom — 自定义校验表达式

独立的纯函数，与 `evalDepends` 类似，但支持 `value` 关键字代表当前字段值：

```js
function evalCustom(expr, value, fm, cellMap) {
  // 第1步：替换 value 关键字为当前字段值
  var e = expr.replace(/\bvalue\b/g, function() {
    if (value === null || value === undefined) return 'null';
    if (typeof value === 'number') return String(value);
    if (typeof value === 'string') return "'" + value.replace(/'/g, "\\'") + "'";
    return 'null';
  });
  // 第2步：替换其他 cell id 为 fm 中的值（与 evalDepends 相同）
  var ids = Object.keys(cellMap).sort(function(a, b) {
    return b.length - a.length;
  });
  ids.forEach(function(id) { /* ... 同 evalDepends ... */ });
  // 第3步：求值
  try { return new Function('"use strict"; return (' + e + ')')(); }
  catch(ex) { return false; }
}
```

**值来源说明**：

| 表达式中的标识 | 值来源 | 说明 |
|---|---|---|
| `value` | 当前被校验字段的值 | async-validator 自动传入，即 `fm[当前cell.id]` |
| cell id（如 `bm_r42`） | `fm[cellId]` | 引擎遍历 cellMap 中的 id，从 fm 取值替换 |

### 11.1.4 validator 函数体校验

`validator` 字段是 JS 函数体字符串，引擎预编译为可调用函数：

```js
// buildRules 中的预编译
var compiledFn = new Function('value', 'fm', 'extra', rule.validator);

// 校验时调用
var result = compiledFn(value, self.fm, self.extraContext);
if (!result) callback(new Error(message));
```

**函数签名**：`function(value, fm, extra)`

| 参数 | 来源 | 说明 |
|------|------|------|
| `value` | async-validator 回调参数 | 当前被校验字段值 |
| `fm` | `self.fm` | 表单实时值字典 `{cellId: value}` |
| `extra` | `self.extraContext` | 第三方外部数据，外部通过 `app.extraContext = {...}` 注入 |

**优先级**：同一规则中 `validator` > `custom` > 原始规则（required/min/max/pattern）。

**配置示例**：

```json
{
  "message": "指标值必须在标杆值和基准值之间",
  "validator": "return value != null && value >= fm.bm_r42 && value <= fm.bl_r42;"
}
```

**extraContext 注入**：

```js
// 外部代码注入第三方值
app.extraContext = { minThreshold: 100, userRole: 'admin' };

// 规则中引用
// "validator": "return value >= extra.minThreshold;"
```

**执行流程示例**：

```
表达式: "value >= bm_r42 && value <= bl_r42"
当前字段 indval_r42 的值 = 30

第1步: 替换 value
  → "30 >= bm_r42 && 30 <= bl_r42"

第2步: 替换 cell id
  bm_r42 → fm['bm_r42'] = 2
  bl_r42 → fm['bl_r42'] = 51.1
  → "30 >= 2 && 30 <= 51.1"

第3步: 求值 → true (校验通过)
```

### 11.1.3 upload 文件校验

upload 类型在 `onUploadFile` 中即时校验，不走 async-validator：

```js
onUploadFile: function(id, e) {
  var file = e.target.files[0];
  var cell = this.cellMap[id];
  // 格式校验
  if (cell.accept) {
    var ext = '.' + file.name.split('.').pop().toLowerCase();
    var allowed = cell.accept.toLowerCase().split(',').map(function(s) { return s.trim(); });
    if (allowed.indexOf(ext) < 0) {
      this.$Message.error('不支持的文件格式，仅允许: ' + cell.accept);
      e.target.value = '';
      return;
    }
  }
  // 大小校验
  if (cell.maxSize) {
    var maxBytes = cell.maxSize * 1024 * 1024;
    if (file.size > maxBytes) {
      this.$Message.error('文件大小超过限制，最大 ' + cell.maxSize + 'MB');
      e.target.value = '';
      return;
    }
  }
  // 通过 → 存储
  this.$set(this.fm, id, JSON.stringify({name, size, type, lastModified}));
  this.onChange(id);
}
```

| 校验 | 配置字段 | 校验时机 | 不通过行为 |
|------|---------|---------|-----------|
| 是否必填 | `required` + `depends` | 表单提交时 | 红框提示 |
| 文件格式 | `accept` | 上传时 | `$Message.error` + 拒绝 |
| 文件大小 | `maxSize` | 上传时 | `$Message.error` + 拒绝 |

### 11.2 number 类型校验要点

async-validator 的 `{required:true, type:"number"}` 校验逻辑：
- `undefined` → 校验失败（不能判断为"空"）
- `null` → 校验失败（视为空）
- `0` → 校验通过
- `NaN` → 校验失败

所以 `setVal` 中必须把空值统一为 `null`，不能是 `undefined`。

### 12.1 recalc — 重算所有公式

### 12.2 evalF — 公式求值

```js
evalF: function(f) {
  var self = this, m;

  // SUM(id1,id2,...)
  m = f.match(/^SUM\((.+)\)$/i);
  if (m) {
    var s = 0;
    m[1].split(',').map(function(x){return x.trim()}).forEach(function(id){
      var c = self.cellMap[id];
      if (c) {
        var v = c.type === 'text' ? parseFloat(c.content) : parseFloat(self.fm[id]);
        if (!isNaN(v)) s += v;
      }
    });
    return s;
  }

  // AVG(id1,id2,...)
  m = f.match(/^AVG\((.+)\)$/i);
  if (m) {
    var s2 = 0, n = 0;
    m[1].split(',').map(function(x){return x.trim()}).forEach(function(id){
      var v = parseFloat(self.fm[id]);
      if (!isNaN(v)) { s2 += v; n++; }
    });
    return n ? +(s2 / n).toFixed(2) : 0;
  }

  // CALC_SCORE — 所有 number 控件值求和
  if (f === 'CALC_SCORE') {
    var t = 0;
    self.config.cells.forEach(function(c){
      if (c.type === 'number' && self.fm[c.id] != null) {
        var v = parseFloat(self.fm[c.id]);
        if (!isNaN(v)) t += v;
      }
    });
    return t ? t.toFixed(2) : '';
  }

  // CALC(表达式) — 标准 JS 表达式，支持 Math 对象
  m = f.match(/^CALC\((.+)\)$/i);
  if (m) {
    try {
      var ce = m[1];
      Object.keys(self.cellMap).forEach(function(id){
        var c = self.cellMap[id];
        var v = c.type === 'text' ? parseFloat(c.content) : parseFloat(self.fm[id]);
        if (!isNaN(v)) ce = ce.replace(new RegExp('\\b' + id + '\\b', 'g'), v);
      });
      return new Function('"use strict"; return (' + ce + ')')();
    } catch(x) {
      return 'ERR';
    }
  }

  // 四则运算表达式：将 cell id 替换为数值后求值
  try {
    var e = f;
    Object.keys(self.cellMap).forEach(function(id){
      var v = parseFloat(self.fm[id]);
      if (!isNaN(v)) e = e.replace(new RegExp('\\b' + id + '\\b', 'g'), v);
    });
    if (/^[\d\s+\-*/().]+$/.test(e))
      return new Function('"use strict"; return (' + e + ')')();
  } catch(x) {}

  return f;  // 无法解析，原样返回
}
```

引用规则：
- `text` 类型 cell → 引用 `content` 字段（`parseFloat(c.content)`）
- 其他类型 cell → 引用 `fm[id]`（表单实际值）
- 引用不存在或值为 NaN → 跳过（按 0 处理）

### 12.3 CALC(表达式) 详细说明

`CALC(表达式)` 允许使用标准 JavaScript 表达式进行自定义计算，支持 Math 对象的所有方法。

| 特性 | 示例 |
|------|------|
| 四则运算 | `CALC((c8 - c9) / c10 * 100)` |
| Math 函数 | `CALC(Math.max(0, c8 - c9))` |
| 三元表达式 | `CALC(c8 > c9 ? c8 : c9)` |
| 四舍五入 | `CALC(Math.round(c8 * 100) / 100)` |
| 范围裁剪 | `CALC(c8 <= 0 ? 0 : c8 >= 100 ? 100 : c8)` |

规则：
- 表达式内部是标准 JavaScript 语法
- cell id 会被自动替换为对应数值（text 类型取 content 的数值，其他类型取 fm[id]）
- Math 对象的所有方法可用（Math.max, Math.min, Math.round, Math.floor, Math.ceil, Math.abs, Math.pow 等）
- 表达式中不能嵌套 SUM/AVG 等引擎函数（它们不是 JS 原生函数），只能用标准 JS 语法
- 计算出错时返回 `"ERR"`，渲染为 ERR 文本

## 十三、数据提交与回显

### 13.1 提交

```js
handleSubmit: function() {
  var self = this;
  this.$refs.mainForm.validate(function(valid) {
    if (valid) {
      var result = { title: self.config.title, cells: [] };
      self.config.cells.forEach(function(c) {
        // 跳过纯展示类型
        if (['label','text','formula'].indexOf(c.type) >= 0) return;
        // 跳过当前隐藏或禁用的控件
        if (self.isHidden(c.id) || self.isDisabled(c.id)) return;
        result.cells.push({
          id: c.id,
          type: c.type,
          value: self.fm[c.id]
        });
      });
      console.log('提交数据:', JSON.stringify(result, null, 2));
    }
  });
}
```

### 13.2 回显 loadData

```js
loadData: function(data) {
  var self = this;
  if (!data || !data.cells) return;

  data.cells.forEach(function(item) {
    if (item.id == null) return;
    var cell = self.cellMap[item.id];
    if (!cell) return;

    if (cell.type === 'number') {
      self.$set(self.fm, item.id,
        (item.value != null && item.value !== '') ? Number(item.value) : null);
    } else if (cell.type === 'checkbox') {
      self.$set(self.fm, item.id, Array.isArray(item.value) ? item.value : []);
    } else {
      self.$set(self.fm, item.id, item.value != null ? item.value : '');
    }
  });

  this.recalc();  // 重算公式
}
```

外部调用方式：`app.loadData({cells: [{id:"c8",value:150}, ...]})`

### 13.3 重置

```js
handleReset: function() {
  var self = this;

  // 1. 恢复 fm 初始值
  this.config.cells.forEach(function(c) {
    if (c.type === 'number')
      self.$set(self.fm, c.id, (c.value != null && c.value !== '') ? Number(c.value) : null);
    else if (c.type === 'checkbox')
      self.$set(self.fm, c.id, []);
    else if (['input','select','radio','date'].indexOf(c.type) >= 0)
      self.$set(self.fm, c.id, c.value != null ? c.value : '');
  });

  // 2. 恢复 disabled/hidden 初始状态
  this.disabledState = {};
  this.hiddenState = {};
  this.config.cells.forEach(function(c) {
    if (c.disabled) self.$set(self.disabledState, c.id, true);
    if (c._hidden)  self.$set(self.hiddenState, c.id, true);
  });

  // 3. 重置提交按钮和校验
  this.submitDisabled = false;
  this.$refs.mainForm.resetFields();

  // 4. 重算公式
  this.recalc();
}
```

### 14.2 单元格样式

```css
th, td {

  vertical-align: middle;
  font-size: 13px;
  line-height: 1.5;
}

td { background: #fff; }
tr:hover td { background: #f0f5ff; }
```

### 14.3 iView 组件在 td 内的适配

```css
td .ivu-form-item { margin-bottom: 0; }
td .ivu-form-item-error-tip { position: relative; padding-top: 2px; font-size: 11px; }
td .ivu-input-number { width: 100%; }
td .ivu-select { width: 100%; }
```

- `margin-bottom:0` — iView FormItem 默认有 24px 下边距，在 td 内要去掉
- `position:relative` — 校验提示默认是 absolute 定位，改为 relative 在 td 内正常显示
- `width:100%` — InputNumber 和 Select 默认宽度不够

### 14.4 Radio 换行（踩坑汇总）

iView 4.x 的 RadioGroup vertical 模式，内部 CSS：

```css
/* iView 内置（会被我们的覆盖） */
.ivu-radio-group-vertical .ivu-radio-wrapper {
  display: block;
  height: 30px;          /* ← 罪魁祸首：固定高度 */
  line-height: 30px;
  white-space: nowrap;   /* ← 不换行 */
}
```

我们的覆盖方案：

```css
.radio-in-cell { text-align: left; padding: 2px 0; }
.radio-in-cell .ivu-radio-group {
  display: flex !important;
  flex-direction: column;
  align-items: flex-start;
  gap: 10px;
}
.radio-in-cell .ivu-radio-wrapper {
  white-space: normal !important;
  height: auto !important;
  line-height: 1.5 !important;
  display: block !important;
  margin-right: 0;
}
.radio-in-cell .ivu-radio-wrapper > .ivu-radio {
  vertical-align: top;
  margin-top: 2px;
}
```

必须使用 `!important` 覆盖 iView 内置样式。
