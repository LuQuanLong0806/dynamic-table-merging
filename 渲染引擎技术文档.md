### 2.1 为什么用 `<script type="text/x-template">`

Vue 2 支持三种模板方式：
1. **字符串模板** `template: '<div>...</div>'` — 需要大量转义引号，易出错
2. **inline-template** — 解析时机不可控
3. **x-template** `<script type="text/x-template" id="...">` — **推荐**：浏览器不执行，Vue 通过 `template: '#xxx'` 引用，无需转义

### 2.2 为什么拆 preview-renderer 子组件

早期版本用 iframe 嵌入预览，切换时卡顿明显（见 [memory: 预览渲染子组件而非 iframe](C:/Users/admin/.claude/projects/f--00-project-------/memory/feedback_preview_rendering.md)）。改为同一 Vue 实例树内的子组件：
- 共享响应式数据，零序列化开销
- 切换视角（portal/platform）即时生效
- 子组件可独立 `validate()`，不污染主应用表单状态

### 4.1 parseRange

```js
function parseRange(v) {
  var p = String(v).split(',').map(Number);
  var mn = Math.min.apply(null, p);
  var mx = Math.max.apply(null, p);
  return { start: mn, end: mx, span: mx - mn + 1 };
}
```

示例：`parseRange("3")` → `{start:3, end:3, span:1}`；`parseRange("2,7")` → `{start:2, end:7, span:6}`

### 4.2 buildGrid — 扁平 cells → 二维网格

```js
function buildGrid(cells) {
  var safeCells = cells || [];
  var cellMap = {};
  safeCells.forEach(function (c) { cellMap[c.id] = c; });

  // 计算网格尺寸
  var maxC = 0, maxR = 0;
  safeCells.forEach(function (c) {
    var xr = parseRange(c.x), yr = parseRange(c.y);
    if (xr.end > maxC) maxC = xr.end;
    if (yr.end > maxR) maxR = yr.end;
  });

  // G[r][c]=entry | null；O[r][c]=true 表示被合并占据
  var G = [], O = [];
  for (var r = 0; r < maxR; r++) {
    G[r] = []; O[r] = [];
    for (var c = 0; c < maxC; c++) { G[r][c] = null; O[r][c] = false; }
  }

  // 关键：按合并面积从大到小排序，大合并先放置，避免被小 cell 拆散
  var sorted = safeCells.slice().sort(function (a, b) {
    return parseRange(b.y).span * parseRange(b.x).span -
           parseRange(a.y).span * parseRange(a.x).span;
  });

  sorted.forEach(function (cell) {
    var xr = parseRange(cell.x), yr = parseRange(cell.y);
    var r0 = yr.start - 1, c0 = xr.start - 1;
    // 防御：x/y 非法或越界 → 跳过
    if (!isFinite(r0) || !isFinite(c0) || r0 < 0 || c0 < 0 || r0 >= maxR || c0 >= maxC) return;
    var entry = { cell: cell, rowspan: yr.span, colspan: xr.span, alts: null };

    // 同位置已有 cell → 当前作为备选项(alts)追加（见 §4.5）
    if (G[r0][c0]) {
      if (!G[r0][c0].alts) G[r0][c0].alts = [];
      G[r0][c0].alts.push(entry);
      return;
    }

    // 占据主位 + 标记合并覆盖区
    G[r0][c0] = entry;
    for (var r = r0; r < r0 + yr.span && r < maxR; r++)
      for (var c = c0; c < c0 + xr.span && c < maxC; c++)
        if (!(r === r0 && c === c0)) O[r][c] = true;
  });

  // 生成渲染用的行数组
  var rows = [];
  for (var r = 0; r < maxR; r++) {
    var cols = [];
    for (var c = 0; c < maxC; c++) {
      if (O[r][c]) { cols.push({ skip: true }); continue; }
      var e = G[r][c];
      if (!e) { cols.push({ skip: false, cell: null }); continue; }
      var isCtrl = ['input','textarea','number','select','radio','checkbox','date','upload']
                   .indexOf(e.cell.type) >= 0;
      cols.push({
        skip: false, cell: e.cell, rowspan: e.rowspan, colspan: e.colspan, alts: e.alts || [],
        isLabel: e.cell.type === 'label', isText: e.cell.type === 'text',
        isControl: isCtrl, isFormula: e.cell.type === 'formula', cls: ...
      });
    }
    rows.push(cols);
  }
  return { rows: rows, cellMap: cellMap, maxC: maxC, maxR: maxR };
}
```

**算法图解**：

```
cell A: x:"1", y:"1,3" (面积=3)
cell B: x:"2,3", y:"1"  (面积=2)
cell C: x:"2", y:"2"
cell D: x:"3", y:"2"

排序：A(3) → B(2) → C(1) → D(1)

放置 A 后：       放置 B 后：       放置 C/D 后：
┌───┬───┬───┐    ┌───┬───┬───┐    ┌───┬───┬───┐
│ A │   │   │    │ A │  B │  B │    │ A │  B │  B │
│   ├───┼───┤    │   ├───┴───┤    │   ├───┼───┤
│   │   │   │    │   │       │    │   │ C │ D │
│   ├───┼───┤    │   │       │    │   ├───┼───┤
│   │   │   │    │   │       │    │   │   │   │
└───┴───┴───┘    └───┴───────┘    └───┴───┴───┘
```

### 4.3 buildFormModel — cells → 表单值字典

```js
function buildFormModel(cells) {
  var m = {};
  (cells || []).forEach(function (c) {
    if (!c || !c.id) return;
    if (c.type === 'number') {
      m[c.id] = c.value != null && c.value !== '' ? Number(c.value) : null;
    } else if (c.type === 'checkbox') {
      m[c.id] = Array.isArray(c.value) ? c.value : [];
    } else if (c.type === 'upload') {
      // upload 多文件累积，统一数组 [{path, name}, ...]
      m[c.id] = Array.isArray(c.value) ? c.value : [];
    } else if (['input', 'textarea', 'select', 'radio', 'date'].indexOf(c.type) >= 0) {
      m[c.id] = c.value != null ? c.value : '';
    }
    // label/text/formula 不初始化（不参与表单），formula 由 recalc 写入
  });
  return m;
}
```

| type | 无 value 初始值 | 有 value 处理 |
|---|---|---|
| number | `null` | `Number(value)` |
| checkbox / upload | `[]` | 数组 |
| input / textarea / select / radio / date | `''` | 原样 |
| label / text / formula | 不初始化 | — |

### 4.4 normalizeUploadVal — upload 值归一化

```js
function normalizeUploadVal(val) {
  if (val == null || val === '') return [];
  if (typeof val === 'string') {
    try {
      var parsed = JSON.parse(val);
      if (Array.isArray(parsed)) return parsed;
      if (parsed && typeof parsed === 'object') return [parsed];
      return [];
    } catch (e) { return []; }
  }
  if (Array.isArray(val)) return val;
  if (typeof val === 'object') return [val];
  return [];
}
```

## 六、列宽算法（与生产对齐）

### 6.1 FIXED_COL_RULES

```js
var FIXED_COL_RULES = [
  { p: /^(序号|编号|No\.?)$/i, w: 4 },
  { p: /^(一级|二级|三级)指标$/, w: 6 },
  { p: /^评分规则$/, w: 6 },
  { p: /^(分值|得分)$/, w: 5 }
];
```

### 6.2 列宽计算流程

```
1. 扫描所有可见的 type=label cell（不限 y，不依赖 freezeRows）
2. 按 y 起始行升序排序：让"具体行（y 大）后处理，覆盖占位行（y 小）"
3. 每个 label cell 按 x span 分摊：
   for ci in (xr.start-1 .. xr.end-1):
     if cell.width > 0:       fixedCols[ci] = cell.width / span
     else if content 匹配 FIXED_COL_RULES: fixedCols[ci] = w / span
4. 没被任何 label cell 覆盖的列 → 默认权重 DEFAULT_COL_WEIGHT=5
5. 隐藏列权重设 0
6. 归一化：所有可见列权重 / sum × 100% → 列宽百分比数组
```

**关键修正案例（2026-06-27）**：47-光伏 JSON 的 y=1 title 跨 12 列无 width，y=2 11 个具体表头 spanSum 同为 12。旧算法"选主行"取 y 较小 → y=2 width 全失效。新算法不再选主行，按 y 升序遍历让具体行覆盖占位行。

### 6.3 列隐藏算法

```
对每列 ci：
  colHasAnyCell[ci]         = 列上是否有任何 cell（含合并 cell 跨越）
  colHasSingleCell[ci]      = 列上是否有独立 cell（span=1）
  colHasVisibleSingleCell[ci] = 列上的独立 cell 是否有可见的

隐藏判断：
  - !colHasAnyCell       → 不隐藏（空列，渲染为 empty-cell-slot）
  - colHasSingleCell     → !colHasVisibleSingleCell （独立 cell 全 hide → 隐藏）
  - 只有合并 cell 跨越    → 不隐藏（让 colspan 动态调整，见 §6.4）
```

> 此修复解决"合并 cell 跨越导致列无法隐藏"的 bug：旧逻辑里合并 cell 也贡献 colHasVisibleCell，导致独立 cell 全 hide 时仍判断为可见，出现空白列。

### 6.4 合并 cell 的 colspan 动态调整

当独立 cell 全 hide 但合并 cell 跨越该列时：
- 列不真正隐藏（`<col>` 仍渲染）
- 但合并 cell 的 colspan 按可见列数动态计算（`visibleColspan`）

## 八、事件系统

### 8.1 runCellEvent — 统一事件入口

```js
runCellEvent: function (eventType, id, value, rawEvent) {
  var cell = this.cellMap[id];
  if (!cell || !cell.events) return;
  var key = 'on' + eventType.charAt(0).toUpperCase() + eventType.slice(1);
  var body = cell.events[key] || cell.events[eventType];
  if (!body || !String(body).trim()) return;

  // 编译缓存（cellMap 不变时 body 不变）
  var cacheKey = eventType + ':' + body;
  var fn = this.eventFnCache[cacheKey];
  if (!fn) {
    try { fn = new Function('value', 'fm', 'extra', 'rawEvent', body); }
    catch (e) { console.error('[事件编译失败]', id, eventType, e); return; }
    this.eventFnCache[cacheKey] = fn;
  }

  try {
    fn.call(this, value, this.fm,
            { cell: cell, cellMap: this.cellMap, envFlag: this.envFlag },
            rawEvent);
  } catch (e) {
    console.error('[事件执行失败]', id, eventType, e);
  }
}
```

### 8.2 函数体 4 个参数

| 参数 | 来源 | 说明 |
|---|---|---|
| `value` | 当前字段值（已 `$set(fm,id,val)` 后） | 可直接用 `value` 或 `fm.当前字段ID` |
| `fm` | `this.fm` | 整个表单 model，**可读可写** |
| `extra` | `{ cell, cellMap, envFlag }` | cell=配置对象，cellMap=id→cell 字典，envFlag='portal'/'platform' |
| `rawEvent` | iView 原参 | input-number/select/radio/checkbox=选中值；date={date:Date, label}；input on-blur=FocusEvent；input onInput=输入字符串 |

### 8.3 三种事件

| 事件 | 触发时机 | 适用类型 | 调用入口 |
|---|---|---|---|
| onChange | 值变化后 | 全部可交互 | `onChange(id, rawEvent)` → `runCellEvent('change', ...)` |
| onBlur | 失焦时 | input / textarea | `onBlur(id, rawEvent)` → `runCellEvent('blur', ...)` |
| onInput | 输入中（每字符） | input / textarea | `onInput(id, rawEvent)` → `runCellEvent('input', ...)` |

### 8.4 写法示例

```js
// 跨字段重算（onChange）
fm.total = (Number(fm.a) || 0) + (Number(fm.b) || 0)

// 格式化（onBlur）
fm.indval_r3 = (fm.indval_r3 || '').trim()

// date-picker 拿 Date 对象（onChange）
fm.date_str = rawEvent.date ? rawEvent.date.toISOString() : ''

// 按端分支（onChange）
if (extra.envFlag === 'portal') fm.score = value * 0.5
else fm.score = value * 1.0
```

### 8.5 onChange 调用链顺序（重要）

```js
onChange: function (id, rawEvent) {
  this.runCellEvent('change', id, this.fm[id], rawEvent);  // ① 用户 JS（可能改其他 cell）
  this.handleLinkage(id);                                  // ② linkage（可能改其他 cell）
  this.recalc();                                           // ③ 公式重算（读最新值）
  this.validateField(id);                                  // ④ 当前字段校验
  this.validateDependents(id);                             // ⑤ 依赖此 cell 的字段重校验
}
```

> 顺序很重要：events/linkage 必须先于 recalc，否则 SUM 拿到上一轮值（见 [memory: onChange 顺序](C:/Users/admin/.claude/projects/f--00-project-------/memory/project_onchange_order_formula.md)）。

### 9.3 为什么不用 v-model

iView 4.x 用 v-model 时：
1. InputNumber 清空时收到 `undefined` 而非 `null`，async-validator `type:"number"` 校验异常
2. Select clearable 清空时收到 `undefined`，Vue 2 检测不到属性变化
3. 响应式丢失

统一用 `:value` + `@on-change`/`@input`，通过 `setVal`/`onNumberChange` 入口写 fm。

### 9.5 onNumberChange — number 专用（增量优化）

```js

  if (cell && cell.type === 'number') {
    val = val === '' || val === undefined || val === null ? null : Number(val);
  }
  this.$set(this.fm, id, val);
  var fi = this._fieldMap[id];
  if (fi) fi.validateState = '';

  var hasEvents = cell && cell.events && (cell.events.onChange || cell.events.change);
  this.runCellEvent('change', id, this.fm[id], val);
  this.handleLinkage(id);
```

### 11.1 buildRules — cell.validation.rules → async-validator 规则

```js
buildRules: function (cell) {
  var v = cell.validation;
  if (!v) return [];
  var isNum = cell.type === 'number';
  var isCheckbox = cell.type === 'checkbox';
  var isUpload = cell.type === 'upload';
  var isArrayType = isCheckbox || isUpload;
  var trigger = isNum || cell.type === 'select' || cell.type === 'radio' || isUpload
                ? 'change' : 'blur';

  if (v.rules && v.rules.length) {
    return v.rules.map(function (r) {
      var rule = {}; for (var k in r) rule[k] = r[k];
      if (!rule.trigger) rule.trigger = trigger;
      if (isNum && rule.required && !rule.validator && !rule.depends && !rule.custom)
        rule.type = rule.type || 'number';
      if (isArrayType && rule.required && !rule.type) rule.type = 'array';

      // string 形式的 required 当作 depends 表达式
      if (typeof rule.required === 'string') {
        rule.depends = rule.required;
        rule.required = true;
      }

      // upload type:array + min：自动构建数组长度校验
      if (isUpload && rule.type === 'array' && rule.min != null && !rule.depends && !rule.validator && !rule.custom) {
        var minVal = rule.min, msg = rule.message || ('至少添加' + minVal + '条数据');
        rule.validator = (function (min, m) {
          return function (r, value, callback) {
            // 空值跳过 min：必填性由 required 规则控制（见 memory: upload_min_skip_empty）
            var isEmpty = false;
            if (Array.isArray(value)) isEmpty = value.length === 0;
            else if (typeof value === 'string') isEmpty = !value.trim();
            else if (value == null) isEmpty = true;
            if (isEmpty) { callback(); return; }
            var ok = Array.isArray(value) ? value.length >= min : /* ... 兼容字符串 */ false;
            if (ok) callback(); else callback(new Error(m));
          };
        })(minVal, msg);
        delete rule.type; delete rule.min;
        return rule;
      }

      // depends/custom/validator：编译为统一 validator
      if (rule.depends || rule.custom || rule.validator) {
        var depExpr = rule.depends || null;
        var customExpr = rule.custom || null;
        var validatorBody = rule.validator || null;
        var orig = { required, type, min, max, pattern, len, enum, message };

        rule.validator = (function (dep, custom, vfn, o) {
          var compiledFn = null;
          if (vfn) try { compiledFn = new Function('value', 'fm', 'extra', vfn); } catch (ex) {}
          return function (r, value, callback) {
            if (dep && !pr_evalDepends(dep, self.fm, self.extraContext)) { callback(); return; }
            if (compiledFn) {
              try { if (!compiledFn(value, self.fm, self.extraContext)) { callback(new Error(o.message)); return; } }
              catch (ex) { callback(new Error(o.message)); return; }
            }
            if (custom && !pr_evalCustom(custom, value, self.fm, self.extraContext)) {
              callback(new Error(o.message)); return;
            }
            if (o.required) {
              var empty = value == null || value === '' || (Array.isArray(value) && value.length === 0);
              if (empty) { callback(new Error(o.message)); return; }
            }
            if (o.type === 'number' && value != null && value !== '') { /* min/max 校验 */ }
            if (o.pattern) { /* 正则校验 */ }
            if (o.len != null && String(value).length !== o.len) { callback(new Error(o.message)); return; }
            if (o.enum && o.enum.indexOf(value) === -1) { callback(new Error(o.message)); return; }
            callback();
          };
        })(depExpr, customExpr, validatorBody, orig);
        // 删除与 validator 冲突的字段
        delete rule.required; delete rule.type; delete rule.min; delete rule.max;
        delete rule.pattern; delete rule.len; delete rule.enum;
        delete rule.message; delete rule.depends; delete rule.custom;
      }
      return rule;
    });
  }

  // 格式2: 简写 {required, pattern, message}
  var rules = [];
  var defaultType = isNum ? 'number' : isArrayType ? 'array' : 'string';
  if (v.required) rules.push({ required: true, message: v.message || '此项为必填', trigger: trigger, type: defaultType });
  if (v.pattern) rules.push({ pattern: new RegExp(v.pattern), message: v.message || '格式不正确', trigger: trigger });
  return rules;
}
```

### 12.2 evalF — 公式求值

```js
evalF: function (f) {
  // ── SUM(id1,id2,...)：求和，text 取 content 数字，其他取 fm[id]，保留 2 位小数 ──
  var m;
  if (m = f.match(/^SUM\((.+)\)$/i)) {
    var s = 0;
    m[1].split(',').forEach(function (id) {
      var c = self.cellMap[id];
      if (c) {
        var v = c.type === 'text' ? parseFloat(c.content) : parseFloat(self.fm[id]);
        if (!isNaN(v)) s += v;
      }
    });
    return +s.toFixed(2);
  }

  // ── AVG ──
  if (m = f.match(/^AVG\((.+)\)$/i)) {
    var s2 = 0, n = 0;
    m[1].split(',').forEach(function (id) {
      var v = parseFloat(self.fm[id]);
      if (!isNaN(v)) { s2 += v; n++; }
    });
    return n ? +(s2 / n).toFixed(2) : 0;
  }

  // ── MIN ──
  if (m = f.match(/^MIN\((.+)\)$/i)) {
    var vals = [];
    m[1].split(',').forEach(function (id) {
      var v = parseFloat(self.fm[id]); if (!isNaN(v)) vals.push(v);
    });
    return vals.length ? Math.min.apply(null, vals) : 0;
  }

  // ── CALC_SCORE：所有 number cell 求和 ──
  if (f === 'CALC_SCORE') {
    var t = 0;
    self.config.cells.forEach(function (c) {
      if (c.type === 'number' && self.fm[c.id] != null) {
        var v = parseFloat(self.fm[c.id]); if (!isNaN(v)) t += v;
      }
    });
    return t ? t.toFixed(2) : '';
  }

  // ── CALC(expr)：标准 JS 表达式，支持 Math.* ──
  if (m = f.match(/^CALC\((.+)\)$/i)) {
    var expr = m[1];
    // 编译缓存
    var compiled = self.formulaCache[expr];
    if (!compiled) {
      var usedIds = Object.keys(self.cellMap).filter(function (id) {
        return new RegExp('\\b' + id + '\\b').test(expr);
      });
      compiled = {
        ids: usedIds,
        fn: new Function(usedIds.concat(['Math']), '"use strict"; return (' + expr + ');')
      };
      self.formulaCache[expr] = compiled;
    }
    // 任一依赖值缺失 → 'ERR'
    var missing = false;
    var args = compiled.ids.map(function (id) {
      var v = parseFloat(self.fm[id]);
      if (isNaN(v)) { missing = true; return 0; }
      return v;
    });
    if (missing) return 'ERR';
    args.push(Math);
    try { return compiled.fn.apply(null, args); }
    catch (x) { return 'ERR'; }
  }

  // ── 兜底：四则运算表达式（cell id 替换为数值后求值） ──
  try {
    var e = f;
    Object.keys(self.cellMap).forEach(function (id) {
      var v = parseFloat(self.fm[id]);
      if (!isNaN(v)) e = e.replace(new RegExp('\\b' + id + '\\b', 'g'), v);
    });
    if (/^[\d\s+\-*/().]+$/.test(e)) return new Function('"use strict"; return (' + e + ')')();
  } catch (x) {}
  return f;  // 无法解析，原样返回
}
```

### 12.5 triggerAllOnChange — 初始化批量触发

```js
triggerAllOnChange: function (reason) {
  if (!this.config || !this.config.cells) return;
  // 找出所有有 onChange 的 cell
  var candidates = this.config.cells.filter(function (c) {
    return c.events && (c.events.onChange || c.events.change);
  });
  // 按 y 升序、x 升序（自然阅读顺序）
  candidates.sort(function (a, b) {
    var ya = parseRange(a.y).start, yb = parseRange(b.y).start;
    if (ya !== yb) return ya - yb;
    return parseRange(a.x).start - parseRange(b.x).start;
  });
  var self = this;
  candidates.forEach(function (c) {
    self.runCellEvent('change', c.id, self.fm[c.id], null);
    self.handleLinkage(c.id);
  });
  this.recalc();  // 末尾统一 recalc 一次
}
```

**reason 标签**：
- `'init-new'`：preview-renderer 初次渲染（让 fm.b = fm.a * 2 这种初始联动跑一次）
- `'init-echo'`：详情回显后（生产组件用，跑一遍初始联动）
- `'rebuild'`：config 变化导致重建

### 13.2 onUploadFile

```js
onUploadFile: function (id, e) {
  var files = e.target.files;
  if (!files || !files.length) return;
  var cell = this.cellMap[id];
  var max = cell && cell.maxCount > 0 ? Number(cell.maxCount) : null;
  var list = Array.isArray(this.fm[id]) ? this.fm[id].slice() : [];
  var accepted = [];

  for (var fi = 0; fi < files.length; fi++) {
    var file = files[fi];
    // accept 后缀校验
    if (cell.accept) {
      var ext = '.' + file.name.split('.').pop().toLowerCase();
      var allowed = cell.accept.toLowerCase().split(',').map(function (s) { return s.trim(); });
      if (allowed.indexOf(ext) < 0) {
        this.$Message.error('不支持的文件格式，仅允许: ' + cell.accept);
        continue;
      }
    }
    // maxCount 上限校验
    if (max != null && list.length + accepted.length >= max) {
      this.$Message.warning('最多上传 ' + max + ' 个文件');
      break;
    }
    accepted.push({ name: file.name, size: file.size, type: file.type, lastModified: file.lastModified });
  }

  if (accepted.length) {
    this.$set(this.fm, id, list.concat(accepted));
    this.onChange(id);
  }
  e.target.value = '';  // 允许下次选同一文件
}
```

## 十九、操作编辑 → 函数调用链 → 渲染刷新路径

> 本节按"用户做了什么"追踪到"哪些函数被调用了 → 哪些 state 被改了 → 渲染怎么刷新"。
> 配置页主应用 = app Vue 实例；preview-renderer = 预览子组件；cell-card = 编辑抽屉里的子组件。

### 19.1 顶部 freezeRows InputNumber 改动

```
用户改 freezeRows
  → v-model 写 config.freezeRows
  → computed 无依赖（freezeRows 不是 preview-renderer 的 watch 触发条件之一?）
  
实际：preview-renderer.watch.config = 'rebuildFromConfig'
  → config 是 prop，父组件改 config.freezeRows 后，子组件收到的 config 引用未变 → watch.config 不触发！
  → 但 freezeRowCount 是 computed，依赖 this.config.freezeRows → 重新计算 → 触发依赖 freezeRowCount 的 DOM 重渲
```

> ⚠️ **隐藏问题**：直接改 `config.freezeRows` 不触发 `watch.config`（引用未变）。只有 `forceRefreshPreview` / `schedulePreview` 会调 `_stripInternal()` 生成新对象再赋给 `previewConfig` → 才触发 `watch.config`。
> **后果**：用户改 freezeRows 后，必须等 `schedulePreview` 防抖（400ms）才会触发 rebuildFromConfig 重建 hiddenState/列宽。期间 freezeRowCount 已变（吸顶行数立即变），但其他依赖 config 的状态没跟上。
> **修复**：在主应用里把 freezeRows 改动也走 `schedulePreview`（实际上 v-model + cell-card `@change` 已经会触发，但顶部 InputNumber 没绑 @on-change）。

### 19.2 编辑抽屉：每个字段改 → 调用链

```
用户在 cell-card 改字段（如 width InputNumber）
  → setCellWidth(val)
  → this.$set(this.cell, 'width', val)
  → this.$emit('change')
  → app-tpl @change="schedulePreview"
  → schedulePreview：
       if (editDrawerOpen) { editDrawerDirty = true; return; }  ← 草稿模式不刷新预览
       if (autoPreview) previewTimer = setTimeout(previewConfig = _stripInternal(), 400ms)
```

**草稿模式**（编辑抽屉打开时）：
- 字段改动只标记 `editDrawerDirty=true`，不刷新预览
- 用户点"保存"→ `commitEditDrawer()` → draft 写回 cell → `closeEditDrawer()`（editDrawerOpen=false）→ `schedulePreview()` → 走完整 400ms 防抖路径
- 用户点"取消"→ `revertEditDrawer()` → 丢弃 draft，原 cell 没被改

### 19.3 行列操作：调用链

```
用户点"插入列" modal 的"确认"
```

`forceRefreshPreview` vs `schedulePreview` 区别：

| 对比项 | schedulePreview | forceRefreshPreview |

| 触发时机 | 编辑抽屉字段改、列表抽屉 cell-card 改 | 行列操作 modal 确认、应用到整列、撤销/重做、导入 |

| editDrawerOpen 短路 | 是（草稿模式只标 dirty） | 否（无条件刷新） |
| autoPreview 短路 | 是 | 否 |
| localStorage 保存 | 500ms 防抖 | 500ms 防抖（同样） |

### 20.1 🟡 removeColumn / removeRow 不清理 linkage.rowTargets.cols 中的悬空 cell id

**位置**：[dynamic-table-utils.js removeColumn/removeRow](dynamic-table-utils.js) / [配置页面.html L2443-2491](配置页面.html#L2443-L2491)

**现象**：
- `removeColumn(col)` 删除独占该列的 cell 时直接 splice，没遍历其他 cell 的 `linkage.actions[].rowTargets[].cols`。
- `removeRow(row)` 同步了 `rowTargets.row`（行号 ±1）但没清理 `rowTargets.cols`（cell id 列表）。

**后果**：
- 用户先在 c8 配联动 rowDisable，rowTargets.cols=['c8','c9']。
- 然后删除 c9（独占某列的 cell）。
- c8 的联动规则里 cols=['c8','c9'] 仍引用 c9，但 c9 已不存在。
- 用户操作触发 c8 联动 → `applyAction` 遍历 cols → `$set(disabledState, 'c9', true)` 写入一个失效 key。
- 渲染层 cellMap['c9'] 不存在 → disabledState['c9'] 不影响任何 cell，但配置 JSON 多了垃圾字段。

**影响**：不崩，但配置污染。serializeJSON 导出会包含 `cols:['c8','c9']`，让后端/生产端拿到的联动规则带失效 id。

**修复建议**：
```js
function removeColumn(cells, col) {
  var removed = /* 现有逻辑 */;
  // 收集被删 cell 的 id
  var deletedIds = cells.splice 后 collect（需在 splice 前先记下）
  // 清理 linkage.rowTargets.cols
  cells.forEach(function (c) {
    if (!c.linkage) return;
    c.linkage.forEach(lk => lk.actions.forEach(a => {
      if (!a.rowTargets) return;
      a.rowTargets.forEach(rt => {
        rt.cols = rt.cols.filter(cid => deletedIds.indexOf(cid) < 0);
      });
    }));
  });
  return removed;
}
```

### 20.3 🟡 insertRows 同步联动 row 号但不区分"跨越插入点"与"完全在插入点之后"

**位置**：[配置页面.html L2397-2411 _shiftLinkageRows](配置页面.html#L2397-L2411)

```js
function _shiftLinkageRows(cells, beforeRow, count) {
  cells.forEach(c => c.linkage?.forEach(lk => lk.actions?.forEach(a => {
    if (!a.rowTargets) return;
    a.rowTargets.forEach(rt => {
      if (rt.row == null) return;
      if (rt.row >= beforeRow) rt.row = rt.row + count;   ← 仅"完全在插入点之后"才 +count
    });
  })));
}
```

**问题**：联动 `rowTargets.row` 是一个具体行号（不是范围），所以只有"完全在插入点之后"或"等于插入点"两种情况。代码 `>=` 把"等于插入点"的也算作后移，意味着：
- 用户在 row=3 配了联动 rowTargets.row=3。
- 在第 3 行前插入 1 行。
- 联动 rowTargets.row 变成 4。

**是否符合预期**？看用户预期：
- 如果用户期望"联动目标固定指向第 3 行的内容"（即原第 3 行被挤到第 4 行）→ 应该 +1，✅
- 如果用户期望"联动目标固定指向物理位置第 3 行"→ 不应该 +1，❌

代码默认**前者**（语义跟随数据，不跟随位置），这与 insertRows 对 cell.y 的处理一致（cells 的 y 也整体后移）。✅ 设计一致。

**结论**：非 bug，但文档应明确"`rowTargets.row` 跟随数据移动"。

### 20.4 🟢 confirmColVis 按起始列归类（合并 cell 跨列不参与"中间列"切换）

**位置**：[配置页面.html L4193-4204 colList](配置页面.html#L4193-L4204)

```js

  this.config.cells.forEach(function (c) {
    var parts = String(c.x).split(',').map(Number);
```

**现象**：合并 cell `x:"3,5"` 归到第 3 列。用户在第 4 列下拉切换可见性时，这个合并 cell 不参与。

**结论**：设计选择。`confirmColVis` 把"列"视为"由独立 cell 决定的列"，合并 cell 视为属于起始列。这不是 bug，但 UI 应提示用户："此列被合并 cell 跨越，请到起始列（第 X 列）调整可见性"。
